  1. 1. Ubuntu14.04挂载硬盘迁移Mysql
    1. 1.1. 前期准备

Ubuntu14.04挂载硬盘迁移Mysql

前期准备

  1. 设置root 密码（如果已有不用重设）
    进入终端 快捷键Ctrl+ALT+T

    • 输入

    apt-get update

apt-get install mysql-client-core-5.6
apt-get install mysql-client-5.6
apt-get install mysql-server-5.6

ps -ef|grep mysql

netstat -tap |grep mysql

service mysql stop

service mysql start

service mysql restart

df -lh

fdisk -lu

fdisk /dev/sda

mkfs -t ext4 /dev/sda

mount -t ext4 /dev/sda /devdata

cd /
mkdir devdata

df -lh

/dev/sda /Projects ext4 defaults 0 3

mkdir p /devdata/mysql

service mysql stop

ls -l

chown -vR mysql:mysql /devdata/mysql

chmod -vR 700 /devdata/mysql

cp -av /var/lib/mysql/* /devdata/mysql

service apparmor reload
service mysql restart

/etc/init.d/apparmor restart
/etc/init.d/mysql restart

rm -rvf /var/lib/mysql/

mysql -uroot -p123

use mysql;

grant all privileges on . to ‘root’@’%’ identified by ‘123’ with grant option;
plush privileges;
exit;

sudo apt-key adv –keyserver hkp://keyserver.ubuntu.com:80 –recv-keys 3FA7E0328081BFF6A14DA29AA6A19B38D3D831EF
echo “deb http://download.mono-project.com/repo/ubuntu xenial main” | sudo tee /etc/apt/sources.list.d/mono-official.list
sudo apt-get update

sudo apt-get install mono-devel

sudo apt-get install mono-complete
```
